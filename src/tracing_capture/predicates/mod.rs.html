1//! Predicates for [`CapturedSpan`]s and [`CapturedEvent`]s.
2//!
3//! # Overview
4//!
5//! A predicate can be created with the functions from this module:
6//!
7//! - [`level()`] checks the span / event level
8//! - [`name()`] checks the span name
9//! - [`target()`] checks the span / event target
10//! - [`field()`] checks a specific span / event field
11//! - [`message()`] checks the event message
12//! - [`parent()`] checks the direct parent span of an event / span
13//! - [`ancestor()`] checks the ancestor spans of an event / span
14//!
15//! These predicates can be combined with bitwise operators, `&` and `|`.
16//! The [`ScanExt`] trait may be used to simplify assertions with predicates. The remaining
17//! traits and structs are lower-level plumbing and rarely need to be used directly.
18//!
19//! [`CapturedSpan`]: crate::CapturedSpan
20//! [`CapturedEvent`]: crate::CapturedEvent
21//!
22//! # Examples
23//!
24//! ```
25//! # use tracing_core::Level;
26//! # use predicates::ord::eq;
27//! # use tracing_capture::{predicates::*, Storage};
28//! # fn test_wrapper(storage: &Storage) {
29//! // Predicates can be combined using bitwise operators:
30//! let predicate = target([eq("tracing")])
31//!     & name(eq("test_capture"))
32//!     & level(Level::INFO)
33//!     & field("result", 42_i64);
34//! // The resulting predicate can be used with `CapturedExt` trait.
35//! let storage: &Storage = // ...
36//! #   storage;
37//! let _ = storage.scan_spans().first(&predicate);
38//! let _ = storage.scan_events().single(&level(Level::ERROR));
39//!
40//! // ...or converted back to a closure:
41//! let _ = storage.all_spans().filter(into_fn(predicate));
42//! # }
43//! ```

70/// Converts a predicate into an `Fn(_) -> bool` closure.
71///
72/// This can be used in APIs (e.g., [`Iterator`] combinators) that expect a closure
73/// as an argument.
74///
75/// # Examples
76///
77/// ```
78/// # use tracing_core::Level;
79/// # use tracing_capture::{predicates::*, CapturedEvent};
80/// let predicate = into_fn(target("tracing") & level(Level::INFO));
81/// let events: &[CapturedEvent] = // ...
82/// #   &[];
83/// let matching_events = events.iter().copied().filter(predicate);
84/// // Do something with `matching_events`...
85/// ```
86pub fn into_fn<Item>(predicate: impl Predicate<Item>) -> impl Fn(&Item) -> bool {
87    move |variable| predicate.eval(variable)
88}
